# 开发组件

# 第一步在创建组件目录

  • 在packages/smoreDesign 下面创建跟antv相同名称的组件目录
  • 新建index.ts src __test__等文件和文件夹

# 第二步编写index.ts

import Button from './src/index.vue';
Button.install = (app: { component: (arg0: string, arg1: any) => void; }) => {
  app.component('s-button', Button);
  return app;
};
export default Button;

# 第三步编写src/index.vue

  • 在编写主题色之前需要先定义主题色，主题色的定义是在packages/themeConfig/index.ts 文件里面定义

  • 主题色的key值定义规则说明：自定义主题色的格式 组件名称_部位(bg|font|border|shadow)_?状态(Hover|Active|Disabled) 没有状态可以省略

import { ColorsType } from './colors';

/**
 * 说明：自定义主题色的格式 
 * 组件名称_部位(bg|font|border|shadow)_?状态(Hover|Active|Disabled)没有状态可以省略
 *
 * @param colors 色系
 * @param themeConfig 自定义主题颜色
 * @returns 所有主题色
 */
export const getThemeUser = (colors: ColorsType, themeConfig: object) => {
  const { Gray_4, Gray_2, Gray_6 } = colors;
  return {
    // 按钮
    Button_Border_Color: Gray_4,
    Button_Bg_Color: Gray_2,
    Button_Bg_Color_Hover: Gray_4,
    Button_Bg_Color_Active: Gray_6,
    Button_Bg_Color_Disabled: Gray_2,
    // input框

    ...colors,
    ...themeConfig
  };
};
  • 主题色定义好之后可以开发组件代码
<script setup lang="ts">
  import buttonProps from 'ant-design-vue/lib/button/buttonTypes';
  import { useTheme } from '../../../themeConfig/useTheme';

  const props = defineProps({
    ...buttonProps()
  });

  const { theme } = useTheme();
</script>

<template>
  <a-button v-bind="$props">
    <slot></slot>
  </a-button>
</template>

<style lang="less" scoped>
  .ant-btn-primary {
    background: v-bind('theme.Button_Bg_Color');
  }
</style>

# 第四步编写测试用例__test__

import { describe, it, expect } from 'vitest';
import { mount, config } from '@vue/test-utils';
import { Button as AButton } from 'ant-design-vue';
import Button from '../src/index.vue';

config.global.stubs = {
  AButton
};

describe('Button', () => {
  it('test Button', () => {
    const wrapper = mount(Button, { props: { type: 'primary' } });
    expect(wrapper.find('.ant-btn').exists()).toBe(true);
  });
  it('test button props', () => {
    const wrapper = mount(Button, { props: { type: 'primary' } });
    expect(wrapper.find('.ant-btn').exists()).toBe(true);
  });
});

  • 第一步先启动example

  • 第二步把写好的组件在src/components.ts文件里面暴露出去

  • 第三步在examples/src/views/ 下面新建一个目录比如说button目录代表这个是测试button的 然后再新建一个demo/demo1.vue的组件
<script setup lang="ts">
import { createApp } from 'vue';

import SmoreDesign from '../src/index';
import 'ant-design-vue/dist/antd.css';
const app = createApp(App);
app.use(router);
app.use(Antd);
app.use(SmoreDesign);
app.mount('#app');

</script>
<template>

  • 第四步在examples/src/router/index.ts 下面新建路由
import { createRouter, createWebHistory } from 'vue-router';

const router = createRouter({
  history: createWebHistory('/examples'),
  routes: [
    {
      name: '首页',
      path: '/',
      component: () => import('../src/views/button/demo/demo1.vue')
    },
    {
      name: 'Button 组件',
      path: '/button',
      component: () => import('../src/views/button/demo/demo1.vue')
    },
  ]
});

export default router;
